29 static Circuit resynthesised_circuit(
30  Circuit &circ,
31  std::optional<std::function<Circuit(const Circuit &)>> transform = nullptr,
32  bool allow_swaps = true) {
33  if (transform.has_value()) {
34  return (*transform)(circ);
35  } else {
36  // Ensure all recognized Clifford gates are Clifford types:
37  decompose_multi_qubits_CX().apply(circ);
38  decompose_cliffords_std().apply(circ);
39  remove_redundancies().apply(circ);
40 
41  // Convert to tableau and back:
42  const UnitaryTableau tab = circuit_to_unitary_tableau(circ);
43  Circuit newcirc = unitary_tableau_to_circuit(tab);
44 
45  // Simplify:
46  clifford_simp(allow_swaps).apply(newcirc);
47 
48  return newcirc;
49  }
50 }

52 static bool resynthesise_cliffords(
53  Circuit &circ,
54  std::optional<std::function<Circuit(const Circuit &)>> transform = nullptr,
55  bool allow_swaps = true) {
56  bool changed = false;
57  for (const VertexSet &verts :
58  circ.get_subcircuits([](Op_ptr op) { return op->is_clifford(); })) {
59  Subcircuit subcircuit = circ.make_subcircuit(verts);
60  Circuit subc = circ.subcircuit(subcircuit);
61  Circuit newsubc = resynthesised_circuit(subc, transform, allow_swaps);
62  circ.substitute(newsubc, subcircuit);
63  changed = true;
64  }
65  return changed;
66 }

68 Transform clifford_resynthesis(
69  std::optional<std::function<Circuit(const Circuit &)>> transform,
