45 static bool multiq_clifford_match(Circuit &circ, bool allow_swaps) {
46  bool success = false;
47  // map from vertex/port to qubit number
48  std::map<Vertex, unit_set_t> v_to_qb = circ.vertex_unit_map();
49  // map from vertex to its depth from slices/reverse_slices
50  std::map<Vertex, unsigned> v_to_depth = circ.vertex_depth_map();
51  std::map<Vertex, unsigned> v_to_rev_depth = circ.vertex_rev_depth_map();
52  // Analysis complete, we can now go through and actually do the transformation
53  VertexList bin;
54  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
55  if (circ.get_OpType_from_Vertex(v) == OpType::CX &&
56  circ.n_out_edges(v) == 2) {
57  // Walk to next CX on the same pair
58  unit_set_t cx_units = v_to_qb.at(v);
59  qubit_vector_t qbs(cx_units.begin(), cx_units.end());
60  QPathDetailed path0;
61  QPathDetailed path1;
62  bool found_cnot = false;
63  Edge e0 = circ.get_nth_out_edge(v, 0);
64  Vertex v0 = circ.target(e0);
65  port_t port = circ.get_target_port(e0);
66  while (true) {
67  OpType v_type = circ.get_OpType_from_Vertex(v0);
68  if (is_final_q_type(v_type)) {
69  break;
70  } else if (v_type == OpType::CX && v_to_qb.at(v0) == cx_units) {
71  found_cnot = true;
72  break;
73  }
74  path0.push_back({v0, port});
75  e0 = circ.get_next_edge(v0, e0);
76  v0 = circ.target(e0);
77  port = circ.get_target_port(e0);
78  }
79  if (!found_cnot) {
80  continue;
81  }
82  Edge e1 = circ.get_nth_out_edge(v, 1);
83  Vertex v1 = circ.target(e1);
84  while (v1 != v0) {
85  path1.push_back({v1, circ.get_target_port(e1)});
86  e1 = circ.get_next_edge(v1, e1);
87  v1 = circ.target(e1);
88  }
89  unsigned path0_length = path0.size();
90  unsigned path1_length = path1.size();
91  unsigned front0_index = 0;
92  unsigned front1_index = 0;
93  while (front0_index < path0_length &&
94  circ.commutes_with_basis(
95  path0[front0_index].first, Pauli::Z, PortType::Target,
96  path0[front0_index].second)) {
97  front0_index++;
98  }
99  while (front1_index < path1_length &&
100  circ.commutes_with_basis(
101  path1[front1_index].first, Pauli::X, PortType::Target,
102  path1[front1_index].second)) {
103  front1_index++;
104  }
105  if (port == 0) {
106  // CXs have same orientation
107  // We walk back from the end of the path
108  // unsigned is used as this is semantically an index and the check for
109  // -1 functions the same as if it were signed
110  unsigned back0_index = path0_length - 1;
111  while (back0_index != front0_index && back0_index != (unsigned)-1 &&
112  circ.commutes_with_basis(
113  path0[back0_index].first, Pauli::Z, PortType::Target,
114  path0[back0_index].second)) {
115  back0_index--;
116  }
117  unsigned back1_index = path1_length - 1;
118  while (back1_index != front1_index && back1_index != (unsigned)-1 &&
119  circ.commutes_with_basis(
120  path1[back1_index].first, Pauli::X, PortType::Target,
121  path1[back1_index].second)) {
122  back1_index--;
123  }
124  // Check for valid causal ordering
125  Vertex front0pre =
126  (front0_index == 0) ? v : path0[front0_index - 1].first;
127  Vertex front0post =
128  (front0_index == path0_length) ? v0 : path0[front0_index].first;
129  Vertex front1pre =
130  (front1_index == 0) ? v : path1[front1_index - 1].first;
131  Vertex front1post =
132  (front1_index == path1_length) ? v0 : path1[front1_index].first;
133  Vertex back0pre =
134  (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
135  Vertex back0post = (back0_index == path0_length - 1)
136  ? v0
137  : path0[back0_index + 1].first;
138  Vertex back1pre =
139  (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
140  Vertex back1post = (back1_index == path1_length - 1)
141  ? v0
142  : path1[back1_index + 1].first;
143  if (circ.in_causal_order(
144  front1pre, front0post, true, v_to_depth, v_to_qb))
145  continue;
146  if (circ.in_causal_order(
147  front0pre, front1post, true, v_to_depth, v_to_qb))
148  continue;
149  if (circ.in_causal_order(
150  back1post, back0pre, false, v_to_rev_depth, v_to_qb))
151  continue;
152  if (circ.in_causal_order(
153  back0post, back1pre, false, v_to_rev_depth, v_to_qb))
154  continue;
155  // Identify which pattern to replace
156  bool v_on_q0 = false;
157  if (front0_index != path0_length) {
158  if (circ.get_OpType_from_Vertex(path0[front0_index].first) !=
159  OpType::V) {
160  // The only non-empty pattern we can recognise is from a V
161  continue;
162  }
163  v_on_q0 = true;
164  if (back0_index != front0_index) {
165  // There are some ops after the V that we cannot commute past the CX
166  continue;
167  }
168  }
169  bool s_on_q1 = false;
170  if (front1_index != path1_length) {
171  if (circ.get_OpType_from_Vertex(path1[front1_index].first) !=
172  OpType::S) {
173  // The only non-empty pattern we can recognise is from a S
174  continue;
175  }
176  s_on_q1 = true;
177  if (back1_index == front1_index + 2 &&
178  circ.get_OpType_from_Vertex(path1[front1_index + 1].first) ==
179  OpType::V &&
180  circ.get_OpType_from_Vertex(path1[front1_index + 2].first) ==
181  OpType::S) {
182  /* --C-- ... --C-- --C-- ... --C-- */
183  /* | | => | | */
184  /* --X--S--V--S--X-- --X--V--S--V--X-- */
185  circ.dag[path1[front1_index].first] = {get_op_ptr(OpType::V)};
186  circ.dag[path1[front1_index + 1].first] = {get_op_ptr(OpType::S)};
187  circ.dag[path1[front1_index + 2].first] = {get_op_ptr(OpType::V)};
188  circ.add_phase(0.25);
189  front1_index++;
190  back1_index--;
191  } else if (back1_index != front1_index) {
192  // There are some ops after the S that we cannot commute past the CX
193  continue;
194  }
195  }
196  if (!allow_swaps && v_on_q0 && s_on_q1) {
197  continue;
198  }
199  success = true;
200  // Always remove the CX pair, saving the new edges from the second
201  Edge next_e0 = circ.get_nth_out_edge(v0, 0);
202  Edge next_e1 = circ.get_nth_out_edge(v0, 1);
203  Vertex next_v0 = circ.target(next_e0);
204  Vertex next_v1 = circ.target(next_e1);
205  port_t next_p0 = circ.get_target_port(next_e0);
206  port_t next_p1 = circ.get_target_port(next_e1);
207  bin.push_back(v);
208  bin.push_back(v0);
209  circ.remove_vertex(
210  v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
211  circ.remove_vertex(
212  v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
213  Edge default_h0 = circ.get_nth_in_edge(next_v0, next_p0);
214  Edge default_h1 = circ.get_nth_in_edge(next_v1, next_p1);
215 
216  Subcircuit sub;
217  const Circuit *replacement;
218  Qubit q0, q1;
219  if (v_on_q0) {
220  Vertex v_gate = path0[front0_index].first;
221  q0 = Qubit(*v_to_qb.at(v_gate).begin());
222  q1 = qbs.at(0);
223  if (q0 == q1) q1 = qbs.at(1);
224  if (s_on_q1) {
225  /* --C--V--C-- --Z--S--V--X--S--\ /-- */
226  /* | | => | \ */
227  /* --X--S--X-- --X--V--S--C--V--/ \-- */
228  Vertex s_gate = path1[front1_index].first;
229  bin.push_back(v_gate);
230  bin.push_back(s_gate);
231  sub = {
232  {circ.get_nth_in_edge(v_gate, 0),
233  circ.get_nth_in_edge(s_gate, 0)},
234  {circ.get_nth_out_edge(v_gate, 0),
235  circ.get_nth_out_edge(s_gate, 0)},
236  {v_gate, s_gate}};
237  replacement = &CircPool::CX_VS_CX_reduced();
238  } else {
239  /* --C--V--C-- --X--V--S--V--C--V--S-- */
240  /* | | => | */
241  /* --X-----X-- --V-----------X-------- */
242  bin.push_back(v_gate);
243  sub = {
244  {circ.get_nth_in_edge(v_gate, 0), default_h1},
245  {circ.get_nth_out_edge(v_gate, 0), default_h1},
246  {v_gate}};
247  replacement = &CircPool::CX_V_CX_reduced();
248  }
249  } else if (s_on_q1) {
250  /* --C-----C-- --S-----------C-------- */
251  /* | | => | */
252  /* --X--S--X-- --Z--S--V--S--X--S--V-- */
253  Vertex s_gate = path1[front1_index].first;
254  q1 = Qubit(*v_to_qb.at(s_gate).begin());
255  q0 = qbs.at(0);
256  if (q0 == q1) q0 = qbs.at(1);
257  bin.push_back(s_gate);
258  sub = {
259  {default_h0, circ.get_nth_in_edge(s_gate, 0)},
260  {default_h0, circ.get_nth_out_edge(s_gate, 0)},
261  {s_gate}};
262  replacement = &CircPool::CX_S_CX_reduced();
263  } else {
264  /* --C--C-- ---- */
265  /* | | => */
266  /* --X--X-- ---- */
267  continue;
268  }
269  Vertex b0 = circ.source(sub.q_in_hole[0]);
270  port_t p0 = circ.get_source_port(sub.q_in_hole[0]);
271  Vertex b1 = circ.source(sub.q_in_hole[1]);
272  port_t p1 = circ.get_source_port(sub.q_in_hole[1]);
273  Vertex a0 = circ.target(sub.q_out_hole[0]);
274  Vertex a1 = circ.target(sub.q_out_hole[1]);
275  circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
276  // Scan through hole and add vertices to v_to_qb
277  // and give estimates of depth and rev_depth
278  unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
279  unsigned new_rev_depth =
280  std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
281  Edge ei0 = circ.get_nth_out_edge(b0, p0);
282  Vertex vi0 = circ.target(ei0);
283  while (vi0 != a0) {
284  v_to_qb.insert({vi0, {q0}});
285  v_to_depth[vi0] = new_depth;
286  v_to_rev_depth[vi0] = new_rev_depth;
287  ei0 = circ.get_next_edge(vi0, ei0);
288  vi0 = circ.target(ei0);
289  }
290  Edge ei1 = circ.get_nth_out_edge(b1, p1);
291  Vertex vi1 = circ.target(ei1);
292  while (vi1 != a1) {
293  if (v_to_qb.find(vi1) == v_to_qb.end())
294  v_to_qb.insert({vi1, {q1}});
295  else
296  v_to_qb.at(vi1).insert(q1);
297  v_to_depth[vi1] = new_depth;
298  v_to_rev_depth[vi1] = new_rev_depth;
299  ei1 = circ.get_next_edge(vi1, ei1);
300  vi1 = circ.target(ei1);
301  }
302  } else {
303  // CXs have different orientation
304  unsigned back0_index = path0_length - 1;
305  while (back0_index >= front0_index && back0_index != (unsigned)-1 &&
306  circ.commutes_with_basis(
307  path0[back0_index].first, Pauli::X, PortType::Target,
308  path0[back0_index].second)) {
309  back0_index--;
310  }
311  unsigned back1_index = path1_length - 1;
312  while (back1_index >= front1_index && back1_index != (unsigned)-1 &&
313  circ.commutes_with_basis(
314  path1[back1_index].first, Pauli::Z, PortType::Target,
315  path1[back1_index].second)) {
316  back1_index--;
317  }
318  Vertex front0pre =
319  (front0_index == 0) ? v : path0[front0_index - 1].first;
320  Vertex front0post =
321  (front0_index == path0_length) ? v0 : path0[front0_index].first;
322  Vertex front1pre =
323  (front1_index == 0) ? v : path1[front1_index - 1].first;
324  Vertex front1post =
325  (front1_index == path1_length) ? v0 : path1[front1_index].first;
326  Vertex back0pre =
327  (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
328  Vertex back0post = (back0_index == path0_length - 1)
329  ? v0
330  : path0[back0_index + 1].first;
331  Vertex back1pre =
332  (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
333  Vertex back1post = (back1_index == path1_length - 1)
334  ? v0
335  : path1[back1_index + 1].first;
336  if (circ.in_causal_order(
337  front1pre, front0post, true, v_to_depth, v_to_qb))
338  continue;
339  if (circ.in_causal_order(
340  front0pre, front1post, true, v_to_depth, v_to_qb))
341  continue;
342  if (circ.in_causal_order(
343  back1post, back0pre, false, v_to_rev_depth, v_to_qb))
344  continue;
345  if (circ.in_causal_order(
346  back0post, back1pre, false, v_to_rev_depth, v_to_qb))
347  continue;
348  // Identify which pattern
349  bool vs_on_q0 = false;
350  if (back0_index >= front0_index && back0_index != (unsigned)-1) {
351  if (!(back0_index == front0_index + 1 &&
352  circ.get_OpType_from_Vertex(path0[front0_index].first) ==
353  OpType::V &&
354  circ.get_OpType_from_Vertex(path0[back0_index].first) ==
355  OpType::S)) {
356  // The only non-empty pattern we can recognise is from VS
357  continue;
358  }
359  vs_on_q0 = true;
360  }
361  bool sv_on_q1 = false;
362  if (back1_index >= front1_index && back1_index != (unsigned)-1) {
363  if (!(back1_index == front1_index + 1 &&
364  circ.get_OpType_from_Vertex(path1[front1_index].first) ==
365  OpType::S &&
366  circ.get_OpType_from_Vertex(path1[back1_index].first) ==
367  OpType::V)) {
368  // The only non-empty pattern we can recognise is from SV
369  continue;
370  }
371  sv_on_q1 = true;
372  }
373  if (!allow_swaps && !vs_on_q0 && !sv_on_q1) {
374  continue;
375  }
376  success = true;
377  // Always remove the CX pair, saving the edges to insert the replacement
378  // circuit
379  Vertex before0, before1, after0, after1;
380  port_t pb0, pb1, pa0, pa1;
381  if (front0_index > 0) {
382  before0 = path0[front0_index - 1].first;
383  pb0 = path0[front0_index - 1].second;
384  } else {
385  Edge pre0 = circ.get_nth_in_edge(v, 0);
386  before0 = circ.source(pre0);
387  pb0 = circ.get_source_port(pre0);
388  }
389  if (front1_index > 0) {
390  before1 = path1[front1_index - 1].first;
391  pb1 = path1[front1_index - 1].second;
392  } else {
393  Edge pre1 = circ.get_nth_in_edge(v, 1);
394  before1 = circ.source(pre1);
395  pb1 = circ.get_source_port(pre1);
396  }
397  if (back0_index + 1 < path0_length) {
398  after0 = path0[back0_index + 1].first;
399  pa0 = path0[back0_index + 1].second;
400  } else {
401  Edge post0 = circ.get_nth_out_edge(v0, 1);
402  after0 = circ.target(post0);
403  pa0 = circ.get_target_port(post0);
404  }
405  if (back1_index + 1 < path1_length) {
406  after1 = path1[back1_index + 1].first;
407  pa1 = path1[back1_index + 1].second;
408  } else {
409  Edge post1 = circ.get_nth_out_edge(v0, 0);
410  after1 = circ.target(post1);
411  pa1 = circ.get_target_port(post1);
412  }
413  bin.push_back(v);
414  bin.push_back(v0);
415  circ.remove_vertex(
416  v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
417  circ.remove_vertex(
418  v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
419  VertexSet to_remove;
420  const Circuit *replacement;
421  Qubit q0, q1;
422  if (vs_on_q0) {
423  Vertex v_gate = path0[front0_index].first;
424  q0 = Qubit(*v_to_qb.at(v_gate).begin());
425  q1 = qbs.at(0);
426  if (q0 == q1) q1 = qbs.at(1);
427  if (sv_on_q1) {
428  /* --C--V--S--X-- --V--S-- */
429  /* | | => */
430  /* --X--S--V--C-- --S--V-- */
431  continue;
432  } else {
433  /* --C--V--S--X-- --S-----C--V--S-- */
434  /* | | => | */
435  /* --X--------C-- --Z--S--X--S----- */
436  replacement = &CircPool::CX_V_S_XC_reduced();
437  }
438  bin.push_back(path0[front0_index].first);
439  bin.push_back(path0[back0_index].first);
440  to_remove.insert(path0[front0_index].first);
441  to_remove.insert(path0[back0_index].first);
442  } else if (sv_on_q1) {
443  /* --C--------X-- --X--V--C--V----- */
444  /* | | => | */
445  /* --X--S--V--C-- --V-----X--S--V-- */
446  Vertex s_gate = path1[front1_index].first;
447  q1 = Qubit(*v_to_qb.at(s_gate).begin());
448  q0 = qbs.at(0);
449  if (q0 == q1) q0 = qbs.at(1);
450  replacement = &CircPool::CX_S_V_XC_reduced();
451  bin.push_back(path1[front1_index].first);
452  bin.push_back(path1[back1_index].first);
453  to_remove.insert(path1[front1_index].first);
454  to_remove.insert(path1[back1_index].first);
455  } else {
456  /* --C--X-- --X--\ /-- */
457  /* | | => | \ */
458  /* --X--C-- --C--/ \-- */
459  q0 = qbs.at(0);
460  q1 = qbs.at(1);
461  replacement = &CircPool::CX_XC_reduced();
462  }
463  Subcircuit sub = {
464  {circ.get_nth_out_edge(before0, pb0),
465  circ.get_nth_out_edge(before1, pb1)},
466  {circ.get_nth_in_edge(after0, pa0),
467  circ.get_nth_in_edge(after1, pa1)},
468  to_remove};
469  Vertex b0 = circ.source(sub.q_in_hole[0]);
470  port_t p0 = circ.get_source_port(sub.q_in_hole[0]);
471  Vertex b1 = circ.source(sub.q_in_hole[1]);
472  port_t p1 = circ.get_source_port(sub.q_in_hole[1]);
473  Vertex a0 = circ.target(sub.q_out_hole[0]);
474  Vertex a1 = circ.target(sub.q_out_hole[1]);
475  circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
476  // Scan through hole and add vertices to v_to_qb
477  unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
478  unsigned new_rev_depth =
479  std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
480  Edge ei0 = circ.get_nth_out_edge(b0, p0);
481  Vertex vi0 = circ.target(ei0);
482  while (vi0 != a0) {
483  v_to_qb.insert({vi0, {q0}});
484  v_to_depth[vi0] = new_depth;
485  v_to_rev_depth[vi0] = new_rev_depth;
486  ei0 = circ.get_next_edge(vi0, ei0);
487  vi0 = circ.target(ei0);
488  }
489  Edge ei1 = circ.get_nth_out_edge(b1, p1);
490  Vertex vi1 = circ.target(ei1);
491  while (vi1 != a1) {
492  if (v_to_qb.find(vi1) == v_to_qb.end())
493  v_to_qb.insert({vi1, {q1}});
494  else
495  v_to_qb.at(vi1).insert(q1);
496  v_to_depth[vi1] = new_depth;
497  v_to_rev_depth[vi1] = new_rev_depth;
498  ei1 = circ.get_next_edge(vi1, ei1);
499  vi1 = circ.target(ei1);
500  }
501  }
502  }
503  }
504  if (allow_swaps) {
505  circ.replace_SWAPs();
506  }
507  circ.remove_vertices(

560 static bool singleq_clifford_from_edge(
561  Circuit &circ, Edge &e, VertexList &bin) {
562  VertexSet single_vs;
563  Edge ei = e;
564  Vertex vi = circ.target(ei);
565  // stuff to check if it is already in standard form
566  // 6 = not consumed anything, 5=Z, 4=X, 3=first S, 2=V, 1=second S, 0=other
567  unsigned cliff_last = 6;
568  while (circ.detect_singleq_unitary_op(vi)) {
569  // i.e. we have a single qubit unitary
570  single_vs.insert(vi);
571 
572  // keep track of clifford standard form
573  switch (circ.get_OpType_from_Vertex(vi)) {
574  case OpType::Z: {
575  cliff_last = (cliff_last > 5) ? 5 : 0;
576  break;
577  }
578  case OpType::X: {
579  cliff_last = (cliff_last > 4) ? 4 : 0;
580  break;
581  }
582  case OpType::S: {
583  if (cliff_last > 3)
584  cliff_last = 3;
585  else if (cliff_last == 2)
586  cliff_last = 1;
587  else
588  cliff_last = 0;
589  break;
590  }
591  case OpType::V: {
592  cliff_last = (cliff_last > 2) ? 2 : 0;
593  break;
594  }
595  default: {
596  cliff_last = 0;
597  break;
598  }
599  }
600 
601  ei = circ.get_next_edge(vi, ei);
602  vi = circ.target(ei);
603  }
604  // if the sequence is not in the standard form, replace it
605  if (cliff_last == 0) {
606  Subcircuit s = {{e}, {ei}, single_vs};
607  Circuit sub = circ.subcircuit(s);
608  bool reduced = (decompose_single_qubits_TK1() >> squash_1qb_to_tk1() >>
609  decompose_cliffords_std())
610  .apply(sub);
611  if (reduced) {
612  circ.substitute(sub, s, Circuit::VertexDeletion::No);
613  bin.insert(bin.end(), single_vs.begin(), single_vs.end());
614  return true;
615  }
616  }
617  return false;
618 }

674  in_e = circ.get_nth_in_edge(v, 0);

676  success = true;

717 std::pair<VertexSet, std::vector<MeasureVertices>> get_end_of_circuit_clifford(
718  const Circuit &circ) {
719  // Initialize vector to store MeasureVertices for end-of-circuit Measures
720  std::vector<MeasureVertices> end_of_circuit_measures;
721 
722  // Iterate over Qubit boundaries to find Measure gates
723  for (auto [it, end] =
724  circ.boundary.get<TagType>().equal_range(UnitType::Qubit);
725  it != end; it++) {
726  Vertex q_out = it->out_;
727  Edge last_gate_out_edge = circ.get_nth_in_edge(q_out, 0);
728  Vertex last_gate = circ.source(last_gate_out_edge);
729 
730  // Check if last gate is a Measure gate
731  if (circ.get_OpType_from_Vertex(last_gate) == OpType::Measure) {
732  Edge possible_c_out_in_edge = circ.get_nth_out_edge(last_gate, 1);
733  Vertex possible_c_out = circ.target(possible_c_out_in_edge);
734 
735  // If the Measure is followed by a ClOutput, store the MeasureVertices
736  if (circ.get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
737  Bit b(circ.get_id_from_out(possible_c_out));
738  end_of_circuit_measures.push_back(
739  MeasureVertices(Qubit(it->id_), b, last_gate));
740  }
741  }
742  }
743 
744  // Initialize variables for constructing Clifford circuit
745  VertexSet clifford_vertices;
746  std::map<Edge, Qubit> frontier;
747  VertexSet previous;
748 
749  // Populate frontier map with Measure edges and associated qubits
750  for (const auto &mv : end_of_circuit_measures) {
751  frontier.insert({circ.get_nth_in_edge(mv.measure, 0), mv.qubit});
752  }
753 
754  // Start adding Clifford vertices to set of vertices
755  std::vector<std::pair<OpType, std::vector<Qubit>>> clifford_commands;
756  std::vector<Edge> to_erase;
757  while (!frontier.empty()) {
758  VertexSet current;
759  // Iterate over frontier edges to identify Clifford gates
760  for (const auto &e : frontier) {
761  Vertex source = circ.source(e.first);
762  OpDesc desc = circ.get_OpDesc_from_Vertex(source);
763  // Check if gate is a Clifford gate without classical or boolean inputs
764  if (desc.is_clifford_gate() && desc.n_boolean() == 0 &&
765  desc.n_classical() == 0) {
766  current.insert(source);
767  } else {
768  // Remove non-Clifford gates from frontier
769  to_erase.push_back(e.first);
770  }
771  }
772  for (const Edge &e : to_erase) {
773  frontier.erase(e);
774  }
775  // Check if identical to previous vertices
776  if (current == previous) {
777  break;
778  }
779  // Update previous set with current set
780  previous = current;
781 
782  // Process Clifford gates and update frontier
783  for (const Vertex &v : current) {
784  EdgeVec out_edges = circ.get_all_out_edges(v);
785  std::vector<Edge> edges;
786  std::vector<Qubit> qubits;
787 
788  // Check which edges connect to existing qubits in the frontier
789  for (const Edge &edge : out_edges) {
790  auto it = frontier.find(edge);
791  if (it != frontier.end()) {
792  edges.push_back(it->first);
793  qubits.push_back(it->second);
794  }
795  }
796 
797  // If all outgoing edges connect to frontier qubits, add gate to Clifford
798  // circuit
799  if (qubits.size() == out_edges.size()) {
800  clifford_vertices.insert(v);
801  EdgeVec in_edges = circ.get_in_edges(v);
802 
803  // Update frontier with incoming edges to processed gate
804  for (unsigned i = 0; i < edges.size(); i++) {
805  Edge e = edges[i];
806  port_t source = circ.get_source_port(e);
807  frontier.insert({in_edges[source], qubits[i]});
808  frontier.erase(e);
809  }
810  }
811  }
812  }
813  return std::make_pair(clifford_vertices, end_of_circuit_measures);
814 }

816 Transform push_cliffords_through_measures() {
817  return Transform([](Circuit &circ) {
818  // Extract Clifford and Measure vertices information from provided circuit
819  std::pair<VertexSet, std::vector<MeasureVertices>> clifford_info =
820  get_end_of_circuit_clifford(circ);
821  VertexSet clifford_vertices = clifford_info.first;
822  std::vector<MeasureVertices> measure_vertices = clifford_info.second;
823 
824  // Initialize vectors to track various information
825  std::vector<Qubit> qubits;
826  std::vector<Bit> bits;
827  QubitPauliMap base;
828  VertexSet final_measure_vertices;
829 
830  // Populate qubits, bits, basic measurement operators and final measurement
831  // vertices
832  for (const MeasureVertices &mv : measure_vertices) {
833  qubits.push_back(mv.qubit);
834  bits.push_back(mv.bit);
835  base.insert({mv.qubit, Pauli::I});
836  final_measure_vertices.insert(mv.measure);
837  }
838 
839  // Construct a circuit with Clifford gates
840  Circuit clifford_circuit =
841  circ.subcircuit(circ.make_subcircuit(clifford_vertices));
842  UnitaryRevTableau cliff_tab =
843  circuit_to_unitary_rev_tableau(clifford_circuit);
844 
845  // Generate updated measurement operator for each end of circuit measurement
846  std::list<SpSymPauliTensor> measurement_operators;
847  for (const Qubit &q : qubits) {
848  QubitPauliMap copy = base;
849  copy[q] = Pauli::Z;
850  SpPauliStabiliser sps(copy, 2);
851  measurement_operators.push_back(cliff_tab.get_row_product(sps));
852  }
853 
854  // Mutually diagonalize updated measuement operators
855  Circuit mutual_c = mutual_diagonalise(
856  measurement_operators, std::set<Qubit>(qubits.begin(), qubits.end()),
857  CXConfigType::Snake);
858 
859  // If mutual diagonalisation circuit doesn't improve 2-qubit gate
860  // count then keep circuit as is
861  if (mutual_c.count_n_qubit_gates(2) >=
862  clifford_circuit.count_n_qubit_gates(2)) {
863  return false;
864  }
865 
866  // Remove Clifford and Measure vertices from the original circuit
867  // before adding mutual diagonalisation circuit
868  circ.remove_vertices(
869  clifford_vertices, Circuit::GraphRewiring::Yes,
870  Circuit::VertexDeletion::Yes);
871  circ.remove_vertices(
872  final_measure_vertices, Circuit::GraphRewiring::Yes,
873  Circuit::VertexDeletion::Yes);
874  // add mutual diagonalisation circuit
875  circ.append(mutual_c);
876 
877  // add back in end of circuit measurements
878  TKET_ASSERT(qubits.size() == bits.size());
879  for (unsigned i = 0; i < qubits.size(); i++) {
880  circ.add_measure(qubits[i], bits[i]);
881  }
882 
883  // Add classical logic to permute output measurements to correct result
884  std::string reg_name =
885  circ.get_next_c_reg_name(c_permutation_scratch_name());
886  register_t scratch_r = circ.add_c_register(reg_name, bits.size() + 1);
887  // Convert Bit to vector for ease of indexing and assigning
888  std::vector<Bit> scratch_v;
889  for (const auto &b : scratch_r) {
890  Bit scratch_b = Bit(b.second);
891  scratch_v.push_back(scratch_b);
892  }
893 
894  // We need to collect ClassicalX due to phase correction
895  // and permutation due to Z terms in operator
896  std::vector<Bit> phase_correction;
897 
898  TKET_ASSERT(measurement_operators.size() == bits.size());
899  auto it = measurement_operators.begin();
900  for (unsigned i = 0; i < measurement_operators.size(); ++i, ++it) {
901  // For each measurement operator, we collect the qubits in the string
902  // which have Z terms
903  auto string = it->string;
904  std::vector<Bit> parity_bits;
905  for (unsigned j = 0; j < qubits.size(); j++) {
906  Qubit q = qubits[j];
907  auto jt = string.find(q);
908  if (jt != string.end()) {
909  // If the qubit has a Z term, then it's measurement result
910  // needs to be Xored with the original target qubit of the
911  // measurement
912  if (jt->second == Pauli::Z) {
913  parity_bits.push_back(bits[j]);
914  } else {
915  TKET_ASSERT(jt->second == Pauli::I);
916  }
917  }
918  }
919  // The measurement operator should never be empty, meaning
920  // that parity_bits should always have at least one entry
921  TKET_ASSERT(!parity_bits.empty());
922  // we now add the Xor corresponding to the correction
923  for (const Bit &pb : parity_bits) {
924  std::vector<Bit> arg = {pb, Bit(scratch_v[i])};
925  circ.add_op(XorWithOp(), arg);
926  }
927 
928  // Finally, we check the phase, and store the required Bit
929  // to flip as appropriate
930  if (it->coeff == 1) {
931  phase_correction.push_back(scratch_v[i]);
932  } else {
933  TKET_ASSERT(it->coeff == -1);
934  }
935  }
936 
937  // Apply the constant change to bitstring due to phase
938  circ.add_op(
939  std::make_shared<SetBitsOp>(std::vector<bool>({true})),
940  std::vector<Bit>({scratch_v.back()}));
941  for (const auto &p_bit : phase_correction) {
942  circ.add_op(XorWithOp(), std::vector<Bit>({scratch_v.back(), p_bit}));
943  }
944 
945  // Copy scratch results over to original Bit
946  TKET_ASSERT(bits.size() + 1 == scratch_v.size());
947  scratch_v.pop_back();
948  scratch_v.insert(scratch_v.end(), bits.begin(), bits.end());
949  circ.add_op(std::make_shared<CopyBitsOp>(unsigned(bits.size())), scratch_v);
950  return true;
951  });
