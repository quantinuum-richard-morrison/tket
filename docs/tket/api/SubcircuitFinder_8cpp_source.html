30 // Representation of a connected convex subcircuit.
31 typedef struct {
32  VertexSet verts; // all vertices in the subcircuit
33  VertexSet preds; // all predecessors of vertices in the subcircuit that are
34  // not themselves in it
35  VertexSet succs; // all successors of vertices in the subcircuit that are not
36  // themselves in it
37 } subcircuit_info_t;

39 // Test whether the union of two connected convex subcircuits is connected.
40 static bool union_is_connected(
41  const subcircuit_info_t &subcircuit_info0,
42  const subcircuit_info_t &subcircuit_info1) {
43  const VertexSet &verts0 = subcircuit_info0.verts;
44  const VertexSet &verts1 = subcircuit_info1.verts;
45  for (const Vertex &v : subcircuit_info0.succs) {
46  if (verts1.contains(v)) {
47  return true;
48  }
49  }
50  for (const Vertex &v : subcircuit_info1.succs) {
51  if (verts0.contains(v)) {
52  return true;
53  }
54  }
55  return false;
56 }

58 static VertexSet set_diff(const VertexSet &A, const VertexSet &B) {
59  VertexSet C;
60  for (const Vertex &v : A) {
61  if (!B.contains(v)) {
62  C.insert(v);
63  }
64  }
65  return C;
66 }

68 // Return the union of two disjoint convex connected subcircuits, assuming that
69 // the union is convex and connected.
70 static subcircuit_info_t convex_union(
71  const subcircuit_info_t &subcircuit_info0,
72  const subcircuit_info_t &subcircuit_info1) {
73  const VertexSet &verts0 = subcircuit_info0.verts;
74  const VertexSet &preds0 = subcircuit_info0.preds;
75  const VertexSet &succs0 = subcircuit_info0.succs;
76  const VertexSet &verts1 = subcircuit_info1.verts;
77  const VertexSet &preds1 = subcircuit_info1.preds;
78  const VertexSet &succs1 = subcircuit_info1.succs;
79 
80  // verts = verts0 ∪ verts1
81  VertexSet verts = verts0;
82  verts.insert(verts1.begin(), verts1.end());
83 
84  // preds = (preds0 ∪ preds1) ∖ verts
85  VertexSet preds01 = preds0;
86  preds01.insert(preds1.begin(), preds1.end());
87  VertexSet preds = set_diff(preds01, verts);
88 
89  // succs = (succs0 ∪ succs1) ∖ verts
90  VertexSet succs01 = succs0;
91  succs01.insert(succs1.begin(), succs1.end());
92  VertexSet succs = set_diff(succs01, verts);
93 
94  return {verts, preds, succs};
95 }

97 static std::set<std::pair<Vertex, Vertex>> order_relations(
98  Circuit *circ /*const*/) {
99  // Put the vertices in reverse topological order:
100  VertexVec verts;
101  boost::topological_sort(circ->dag, std::back_inserter(verts));
102  // Construct a map v --> {all vertices in the future of v}:
103  std::map<Vertex, VertexSet> futures;
104  for (const Vertex &v : verts) {
105  VertexSet v_futures = {v};
106  for (const Vertex &w : circ->get_successors(v)) {
107  const VertexSet &w_futures = futures[w];
108  v_futures.insert(w_futures.begin(), w_futures.end());
109  }
110  futures[v] = v_futures;
111  }
112  // Construct the set of order relations:
113  std::set<std::pair<Vertex, Vertex>> rels;
114  for (const auto &v_ws : futures) {
115  const Vertex &v = v_ws.first;
116  for (const Vertex &w : v_ws.second) {
117  rels.insert({v, w});
118  }
119  }
120  return rels;
121 }

123 // Helper class for finding connected convex subcircuits.
124 class SubcircuitFinder {
125  public:
126  SubcircuitFinder(Circuit *circ /*const*/)
127  : circ_(circ), order_relations_(order_relations(circ)) {}
128  std::vector<VertexSet> find_subcircuits(
129  std::function<bool(Op_ptr)> criterion) {
130  // Find a maximal partition of the vertices satisfying the criterion into
131  // connected convex subcircuits. We use a greedy algorithm, beginning with
132  // the trivial partition into singletons, and then repeatedly looking for a
133  // pair that we can merge.
134  std::vector<subcircuit_info_t> subcircuit_infos;
135  // Form a trivial partition, one node per subcircuit:
136  BGL_FORALL_VERTICES(v, circ_->dag, DAG) {
137  if (criterion(circ_->get_Op_ptr_from_Vertex(v))) {
138  VertexVec preds = circ_->get_predecessors(v);
139  VertexVec succs = circ_->get_successors(v);
140  subcircuit_infos.push_back(
141  {{v}, {preds.begin(), preds.end()}, {succs.begin(), succs.end()}});
142  }
143  }
144  while (true) {
145  // If possible, find a mergeable pair of subcircuits and merge them.
146  std::optional<std::pair<std::size_t, std::size_t>> indices =
147  find_mergeable_pair(subcircuit_infos);
148  if (!indices.has_value()) {
149  break;
150  } else {
151  std::size_t i0 = indices->first;
152  std::size_t i1 = indices->second;
153  TKET_ASSERT(i0 < i1);
154  subcircuit_info_t subcircuit_info0 = subcircuit_infos[i0];
155  subcircuit_info_t subcircuit_info1 = subcircuit_infos[i1];
156  // Must erase in this order, since i0 < i1:
157  subcircuit_infos.erase(subcircuit_infos.begin() + i1);
158  subcircuit_infos.erase(subcircuit_infos.begin() + i0);
159  subcircuit_infos.push_back(
160  convex_union(subcircuit_info0, subcircuit_info1));
161  }
162  }
163  std::vector<VertexSet> subcircuits;
164  for (const subcircuit_info_t &subcircuit_info : subcircuit_infos) {
165  subcircuits.push_back(subcircuit_info.verts);
166  }
167  return subcircuits;
168  }
169 
170  private:
171  // Test whether the union of two disjoint connected convex subcircuits is
172  // convex.
173  bool union_is_convex(
174  const subcircuit_info_t &subcircuit_info0,
175  const subcircuit_info_t &subcircuit_info1) {
176  const VertexSet &preds0 = subcircuit_info0.preds;
177  const VertexSet &succs0 = subcircuit_info0.succs;
178  const VertexSet &preds1 = subcircuit_info1.preds;
179  const VertexSet &succs1 = subcircuit_info1.succs;
180  for (const Vertex &v0 : succs0) {
181  for (const Vertex &v1 : preds1) {
182  if (order_relations_.contains({v0, v1})) {
183  return false;
184  }
185  }
186  }
187  for (const Vertex &v1 : succs1) {
188  for (const Vertex &v0 : preds0) {
189  if (order_relations_.contains({v1, v0})) {
190  return false;
191  }
192  }
193  }
194  return true;
195  }

197  // Given a vector of disjoint connected convex subcircuits, look for a pair
198  // whose union is connected and convex, and return the indices of such a pair
199  // if it exists.
200  std::optional<std::pair<std::size_t, std::size_t>> find_mergeable_pair(
201  const std::vector<subcircuit_info_t> &subcircuit_infos) {
202  std::size_t n_subcircuits = subcircuit_infos.size();
203  for (std::size_t i0 = 0; i0 < n_subcircuits; i0++) {
204  for (std::size_t i1 = i0 + 1; i1 < n_subcircuits; i1++) {
205  const subcircuit_info_t &subcircuit_info0 = subcircuit_infos[i0];
206  const subcircuit_info_t &subcircuit_info1 = subcircuit_infos[i1];
207  if (union_is_connected(subcircuit_info0, subcircuit_info1) &&
208  union_is_convex(subcircuit_info0, subcircuit_info1)) {
209  return std::pair<std::size_t, std::size_t>{i0, i1};
210  }
211  }
212  }
213  return std::nullopt;
214  }
215  const Circuit *circ_;
216  std::set<std::pair<Vertex, Vertex>> order_relations_;
217 };

226 Subcircuit Circuit::make_subcircuit(VertexSet verts) const {
227  const std::map<Edge, UnitID> unitmap = edge_unit_map();
228  EdgeSet q_ins, q_outs, c_ins, c_outs;
229  for (const Vertex &v : verts) {
230  for (const Edge &v_in : get_in_edges(v)) {
231  if (!verts.contains(source(v_in))) {
232  switch (get_edgetype(v_in)) {
233  case EdgeType::Quantum:
234  q_ins.insert(v_in);
235  break;
236  case EdgeType::Classical:
237  c_ins.insert(v_in);
238  break;
239  default:
240  // Boolean and WASM edges ignored.
241  // TODO What to do? Throw?
242  break;
243  }
244  }
245  }
246  for (const Edge &v_out : get_all_out_edges(v)) {
247  if (!verts.contains(target(v_out))) {
248  switch (get_edgetype(v_out)) {
249  case EdgeType::Quantum:
250  q_outs.insert(v_out);
251  break;
252  case EdgeType::Classical:
253  c_outs.insert(v_out);
254  break;
255  default:
256  // Boolean and WASM edges ignored.
257  // TODO What to do? Throw?
258  break;
259  }
260  }
261  }
262  }
263  EdgeSet crs;
264  for (const Edge &e : c_outs) {
265  Vertex v = source(e);
266  for (const Edge &b_out : get_out_edges_of_type(v, EdgeType::Boolean)) {
267  crs.insert(b_out);
268  }
269  }
270  // Convert to vectors, taking care of order:
271  TKET_ASSERT(q_ins.size() == q_outs.size());
272  TKET_ASSERT(c_ins.size() == c_outs.size());
273  EdgeVec q_ins_vec{q_ins.begin(), q_ins.end()};
274  EdgeVec q_outs_vec{q_outs.begin(), q_outs.end()};
275  EdgeVec c_ins_vec{c_ins.begin(), c_ins.end()};
276  EdgeVec c_outs_vec{c_outs.begin(), c_outs.end()};
277  EdgeVec crs_vec{crs.begin(), crs.end()};
278  std::sort(
279  q_ins_vec.begin(), q_ins_vec.end(),
280  [&unitmap](const Edge &e0, const Edge &e1) {
281  return unitmap.at(e0) < unitmap.at(e1);
282  });
283  std::sort(
284  q_outs_vec.begin(), q_outs_vec.end(),
285  [&unitmap](const Edge &e0, const Edge &e1) {
286  return unitmap.at(e0) < unitmap.at(e1);
287  });
288  std::sort(
289  c_ins_vec.begin(), c_ins_vec.end(),
290  [&unitmap](const Edge &e0, const Edge &e1) {
291  return unitmap.at(e0) < unitmap.at(e1);
292  });
293  std::sort(
294  c_outs_vec.begin(), c_outs_vec.end(),
295  [&unitmap](const Edge &e0, const Edge &e1) {
296  return unitmap.at(e0) < unitmap.at(e1);
297  });
298  return {q_ins_vec, q_outs_vec, c_ins_vec, c_outs_vec, crs_vec, verts};
299 }
