205 PauliExpCommutingSetBox::PauliExpCommutingSetBox(
206  const std::vector<SymPauliTensor> &pauli_gadgets,
207  CXConfigType cx_config_type)
208  : Box(OpType::PauliExpCommutingSetBox),
209  pauli_gadgets_(pauli_gadgets),
210  cx_config_(cx_config_type) {
211  // check at least one gadget
212  if (pauli_gadgets.empty()) {
213  throw PauliExpBoxInvalidity(
214  "PauliExpCommutingSetBox requires at least one Pauli string");
215  }
216  // check all gadgets have same Pauli string length
217  auto n_qubits = pauli_gadgets[0].size();
218  for (const auto &gadget : pauli_gadgets) {
219  if (gadget.size() != n_qubits) {
220  throw PauliExpBoxInvalidity(
221  "the Pauli strings within PauliExpCommutingSetBox must all be the "
222  "same length");
223  }
224  }
225  if (!this->paulis_commute()) {
226  throw PauliExpBoxInvalidity(
227  "Pauli strings used to define PauliExpCommutingSetBox must all "
228  "commute");
229  }
230  signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
231 }

250 bool PauliExpCommutingSetBox::paulis_commute() const {
251  for (auto string0 = pauli_gadgets_.begin(); string0 != pauli_gadgets_.end();
252  ++string0) {
253  for (auto string1 = std::next(string0); string1 != pauli_gadgets_.end();
254  ++string1) {
255  if (!string0->commutes_with(*string1)) {
256  return false;
257  }
258  }
259  }
260  return true;

300 void PauliExpCommutingSetBox::generate_circuit() const {
301  unsigned n_qubits = pauli_gadgets_[0].size();
302  Circuit circ(n_qubits);
303 
304  std::list<SpSymPauliTensor> gadgets;
305  for (const auto &pauli_gadget : pauli_gadgets_) {
306  gadgets.push_back((SpSymPauliTensor)pauli_gadget);
307  }
308  std::set<Qubit> qubits;
309  for (unsigned i = 0; i < n_qubits; i++) qubits.insert(Qubit(i));
310 
311  Circuit cliff_circ = mutual_diagonalise(gadgets, qubits, cx_config_);
312 
313  Circuit phase_poly_circ(n_qubits);
314 
315  for (const SpSymPauliTensor &pgp : gadgets) {
316  phase_poly_circ.append(pauli_gadget(pgp, CXConfigType::Snake));
317  }
318  phase_poly_circ.decompose_boxes_recursively();
319  PhasePolyBox ppbox(phase_poly_circ);
320  Circuit after_synth_circ = *ppbox.to_circuit();
321 
322  ConjugationBox box(
323  std::make_shared<CircBox>(cliff_circ),
324  std::make_shared<CircBox>(after_synth_circ));
325 
326  circ.add_box(box, circ.all_qubits());
327 
328  circ_ = std::make_shared<Circuit>(circ);
329 }

530 DensePauliMap pad_sparse_pauli_map(
531  const QubitPauliMap &sparse_string, size_t size) {
532  // n.b. as a helper method this can assume that all qubits in sparse string
533  // are labelled with the default qubit register and no index is larger than or
534  // equal to the size
535  std::vector<Pauli> dense_string(size, Pauli::I);
536  for (const auto &letter : sparse_string) {
537  TKET_ASSERT(letter.first.reg_name() == q_default_reg());
538  std::vector<unsigned> index = letter.first.index();
539  TKET_ASSERT(index.size() == 1);
540  TKET_ASSERT(index[0] < size);
541  dense_string[index[0]] = letter.second;
542  }
543  return dense_string;
544 }

547  // constructor guarantees all gadgets are the same size
548  unsigned n_qubits = pauli_gadgets_[0].size();
549  Circuit circ(n_qubits);
550 
551  // Saves later handling
552  if (this->pauli_gadgets_.empty()) {
553  circ_ = std::make_shared<Circuit>(circ);
554  return;
555  }
556 
557  // First combine any PauliGadgets with the same dense string into the same
558  // term
559  // n.b. the map assigns an ordering to the Pauli terms, meaning it
560  // is an algorithm-affecting step
561  std::map<DensePauliMap, Expr> reduced_pauli_gadgets;
562  for (const SymPauliTensor &pauli_gadget : this->pauli_gadgets_) {
563  auto it = reduced_pauli_gadgets.find(pauli_gadget.string);
564  if (it != reduced_pauli_gadgets.end()) {
565  it->second += pauli_gadget.coeff;
566  } else {
567  reduced_pauli_gadgets.insert(
568  std::make_pair(pauli_gadget.string, pauli_gadget.coeff));
569  }
570  }
571 
572  // Construct circuit depending on specified strategy, using other Pauli
573  // Exponential Boxes
574  switch (this->synth_strategy_) {
575  case Transforms::PauliSynthStrat::Individual: {
576  for (auto it = reduced_pauli_gadgets.begin();
577  it != reduced_pauli_gadgets.end(); ++it) {
578  circ.add_box(
579  PauliExpBox(
580  SymPauliTensor(it->first, it->second), this->cx_configuration_),
581  circ.all_qubits());
582  }
583  break;
584  }
585  case Transforms::PauliSynthStrat::Pairwise: {
586  auto it = reduced_pauli_gadgets.begin();
587  // if there is an odd number of gadgets, add first gadget as a single
588  if (reduced_pauli_gadgets.size() % 2 == 1) {
589  circ.add_box(
590  PauliExpBox(
591  SymPauliTensor(it->first, it->second), this->cx_configuration_),
592  circ.all_qubits());
593  ++it;
594  }
595  while (it != reduced_pauli_gadgets.end()) {
596  auto next_it = std::next(it);
597  TKET_ASSERT(next_it != reduced_pauli_gadgets.end());
598  circ.add_box(
599  PauliExpPairBox(
600  SymPauliTensor(it->first, it->second),
601  SymPauliTensor(next_it->first, next_it->second),
602  this->cx_configuration_),
603  circ.all_qubits());
604  std::advance(it, 2);
605  }
606  break;
607  }
608  case Transforms::PauliSynthStrat::Sets: {
609  // term_sequence expects std::list<SpPauliString>, i.e. QubitPauliMap
610  // instead of DensePauliMap first convert the keys of
611  // reduced_pauli_gadgets to a std::list<SpPauliString> object
612  std::list<SpPauliString> all_terms;
613  for (const auto &gadget : reduced_pauli_gadgets) {
614  all_terms.push_back(SpPauliString(gadget.first));
615  }
616  // then get the term sequence into commuting sets
617  std::list<std::list<SpPauliString>> commuting_sets = term_sequence(
618  all_terms, this->partition_strategy_, this->graph_colouring_);
619  // construct each commuting set using PauliExpCommutingSetBox:
620  // 1) pad each QubitPauliMap into a DensePauliMap by adding Pauli.I terms
621  // for missing keys 2) retrieve the exponential coefficient from
622  // reduced_pauli_gadgets 3) construct a SymPauliTensor object as the
623  // argument to PauliExpCommutingSetBox
624  for (const std::list<SpPauliString> &strings : commuting_sets) {
625  std::vector<SymPauliTensor> commuting_gadgets;
626  for (const SpPauliString &string : strings) {
627  DensePauliMap padded_string =
628  pad_sparse_pauli_map(string.string, n_qubits);
629  commuting_gadgets.push_back(SymPauliTensor(
630  padded_string, reduced_pauli_gadgets[padded_string]));
631  }
632  circ.add_box(
633  PauliExpCommutingSetBox(commuting_gadgets, this->cx_configuration_),
634  circ.all_qubits());
635  }
636  break;
637  }
638  default: {
639  throw std::logic_error(
640  "TermSequenceBox passed an unsupported PauliSynthStrat");
641  }
642  }
643  circ_ = std::make_shared<Circuit>(circ);
644 }

661  Circuit &circ, const SpSymPauliTensor &pauli0,
662  const SpSymPauliTensor &pauli1, CXConfigType cx_config) {
663  std::vector<Qubit> mapping;
664  std::vector<Pauli> paulis0;
665  std::vector<Pauli> paulis1;
666  QubitPauliMap p1map = pauli1.string;
667  // add paulis for qubits in pauli0_string
668  for (const std::pair<const Qubit, Pauli> &term : pauli0.string) {
669  mapping.push_back(term.first);
670  paulis0.push_back(term.second);
671  auto found = p1map.find(term.first);
672  if (found == p1map.end()) {
673  paulis1.push_back(Pauli::I);
674  } else {
675  paulis1.push_back(found->second);
676  p1map.erase(found);
677  }
678  }
679  // add paulis for qubits in pauli1_string that weren't in pauli0_string
680  for (const std::pair<const Qubit, Pauli> &term : p1map) {
681  mapping.push_back(term.first);
682  paulis1.push_back(term.second);
683  paulis0.push_back(Pauli::I); // If pauli0_string contained qubit, would
684  // have been handled above
685  }
686  PauliExpPairBox box(
687  SymPauliTensor(paulis0, pauli0.coeff),
688  SymPauliTensor(paulis1, pauli1.coeff), cx_config);
689  circ.add_box(box, mapping);

693  Circuit &circ, const std::list<SpSymPauliTensor> &gadgets,
694  CXConfigType cx_config) {

696  // Preserves ordering of qubits
697 
698  std::set<Qubit> all_qubits;
699  for (const SpSymPauliTensor &gadget : gadgets) {
700  for (const std::pair<const Qubit, Pauli> &qubit_pauli : gadget.string) {
701  all_qubits.insert(qubit_pauli.first);
702  }
703  }
704 
705  std::vector<Qubit> mapping;
706  for (const auto &qubit : all_qubits) {
707  mapping.push_back(qubit);
708  }
709 
710  std::vector<SymPauliTensor> pauli_gadgets;
711  for (const SpSymPauliTensor &gadget : gadgets) {
712  SymPauliTensor &new_gadget =
713  pauli_gadgets.emplace_back(DensePauliMap{}, gadget.coeff);
714  for (const Qubit &qubit : mapping) {
715  new_gadget.string.push_back(gadget.get(qubit));
716  }
717  }
718 
719  PauliExpCommutingSetBox box(pauli_gadgets, cx_config);
720  circ.add_box(box, mapping);
